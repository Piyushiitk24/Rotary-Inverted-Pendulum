  \lhead{\emph{Symbols}}
  \listofnomenclature{lll}{
    $\theta$ & Base or rotary-arm angle & \si{\radian} or \si{\degree} \\
    $\alpha$ & Pendulum angle measured from upright & \si{\radian} or \si{\degree} \\
    $\theta_{\mathrm{ref}}$ & Base-angle reference & \si{\degree} \\
    $u=\ddot{\theta}$ & Abstract commanded base acceleration & \si{\radian\per\second\squared} \\
    $s$ & Sliding variable & controller-dependent \\
    $L_r$ & Rotary-arm length & \si{\metre} \\
    $l_p$ & Pendulum pivot-to-centre-of-mass distance & \si{\metre} \\
    $J_0,J_1$ & Effective arm and pendulum inertias & \si{\kilogram\metre\squared} \\
    $A,B$ & Reduced upright-model constants & \si{\per\second\squared}, dimensionless \\
    $k_s$ & Stepper angle conversion & \si{\step\per\degree} \\
  }
